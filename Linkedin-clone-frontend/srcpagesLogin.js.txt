import { useNavigate } from "react-router-dom";
import "./Login.css";

function Login() {
  const navigate = useNavigate();

  const login = () => {
    localStorage.setItem("user", "Ajay");
    navigate("/feed");
  };

  return (
    <div className="login-container">
      <h1>LinkedIn</h1>
      <button onClick={login}>Sign in</button>
    </div>
  );
}

export default Login;
