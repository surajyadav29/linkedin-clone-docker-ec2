import { useNavigate } from "react-router-dom";
import "./Navbar.css";

function Navbar() {
  const navigate = useNavigate();

  const logout = () => {
    localStorage.clear();
    navigate("/");
  };

  return (
    <div className="navbar">
      <span>LinkedIn</span>
      <button onClick={logout}>Logout</button>
    </div>
  );
}

export default Navbar;
